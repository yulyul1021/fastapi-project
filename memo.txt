<!DOCTYPE html>
<html lang="ko">
<head>
    <meta charset="UTF-8">
    <title>회원가입</title>
</head>
<body>
<script>
        document.addEventListener("DOMContentLoaded", function () {
        // 쿠키에서 access-token 가져오기
        function getCookie(name) {
            var value = document.cookie.match('(^|;) ?' + name + '=([^;]*)(;|$)');
            return value? value[2] : null;
        }

        var accessToken = getCookie('access_token');
        alert(accessToken)

        if (accessToken) {
            // access-token이 존재하면 로그아웃 버튼만 보여줌
            document.getElementById('signin').style.display = '';
            document.getElementById('signup').style.display = '';
            document.getElementById('signout').style.display = 'none';
        } else {
            // access-token이 존재하지 않으면 로그인, 회원가입 버튼만 표시
            document.getElementById('signin').style.display = 'none';
            document.getElementById('signup').style.display = 'none';
            document.getElementById('signout').style.display = '';
         }
    });
</script>
<div class="buttons-container">
        <button class="button" id="signout" onclick="location.href='signout'">로그아웃</button>
        <button class="button" id="signin" onclick="location.href='signin'">로그인</button>
        <button class="button" id="signup" onclick="location.href='signup'">회원가입</button>
</div>
</body>
